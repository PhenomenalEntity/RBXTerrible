function generateRandomString(length, charset)
    charset = charset or "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789~`%_!@#$%^&*()+-{}[]|"
    local result = {}
    local charsetLen = #charset

    for i = 1, length do
        local randomIndex = math.random(1, charsetLen)
        result[i] = string.sub(charset, randomIndex, randomIndex)
    end
    return table.concat(result)
end



while true do 
    writefile(generateRandomString(64), generateRandomString(9999))
    task.wait()
end
